create database união_química

use união_química

create table Usuário (  
	Email varchar(100) primary key not null,     
	Senha varchar(20) not null     
	);

create table fornecedor (
	CNPJ char(18) primary key not null,
    	Nome_Fornecedor varchar(100) not null unique,
    	Email varchar(100) not null,
    	Telefone char(19) not null,
    	Observacoes varchar(255),
    	Endereco varchar(255) not null
    	);

create table Produto (  
	Código int primary key not null auto_increment,
	Nome_Produto varchar(100) not null,
    	Preco_Unitario decimal not null,
    	Quantidade_Estoque int not null,
    	Descricao varchar(255),
    	Fornecedor varchar(100),
    	foreign key (Fornecedor) references fornecedor (Nome_Fornecedor)
    	);

SELECT * FROM união_química.produto LIMIT 0, 1000
